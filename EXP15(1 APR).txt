# RETRIVE AND PRINT FIRST THREE LETTER OF CURRENT MONTH
CREATE OR REPLACE PROCEDURE TO_MONTH
AS
  V_MONTH VARCHAR2(5);
BEGIN
  V_MONTH:=to_char(sysdate,'MON');
  DBMS_OUTPUT.PUT_LINE('THE CURRENT MONTH IS ' || V_MONTH);
END;
/
Procedure created.

SQL> exec to_month;
THE CURRENT MONTH IS APR
PL/SQL procedure successfully completed.

#DESIGN A TRIGGER SUCH THAT NO INSERTION OR UPDATION is done on sat or sun(for practice you ceate trigger for wed)

SQL> CREATE OR REPLACE TRIGGER tri_day
  2  BEFORE INSERT OR UPDATE ON emp2
  3  FOR EACH ROW
  4  DECLARE
  5      v_day VARCHAR2(10);
  6  BEGIN
  7      v_day := UPPER(TO_CHAR(SYSDATE, 'dy'));
  8
  9      IF v_day = 'WED' THEN
 10          RAISE_APPLICATION_ERROR(-20001, 'CANT INSERT ON WEDNESDAY HEHEHE');
 11      END IF;
 12  END;
 13  /

NOTE:
TO_CHAR(SYSDATE, 'day') returns:
