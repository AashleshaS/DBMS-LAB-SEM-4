1. create user :
create user d1 identified by sit;
#user d1 is created with password sit
#user d1 needs permissions for database operations

2. grant permissions :
grant connect, resource to d1;
#permissions granted to user d1

3. change password :
alter user d1 identified by newpassword;
#password can contain only 0-9, a-z, A-Z
#special characters are not allowed

4. login :
enter username as: d1
enter password as: newpassword
#user d1 is now connected to the database

5. commit or exit :
commit;
#saves the changes

OR

exit;
#saves and disconnects from the database
