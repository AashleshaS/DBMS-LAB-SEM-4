METHOD 1-LARGEST OF THREE NUMBERS redirect to prompt

create or replace procedure lar
(a in out number, 
b in out number,
c in out number)
as
begin
  if a>b and a>c then
	DBMS_output.put_line('Largest number is' || a);
  elsif b>a and b>c then
        DBMS_output.put_line('Largest number is' || b);
  else
	DBMS_output.put_line('Largest number is' || c);
End if;
end;
/
Procedure created.

SQL> variable a number;
SQL> exec :a:=10;
PL/SQL procedure successfully completed.

SQL> variable b number;
SQL> exec :b:=20;
PL/SQL procedure successfully completed.

SQL> variable c number;
SQL> exec :c:=30;
PL/SQL procedure successfully completed.

SQL> exec lar(:a,:b,:c);
Largest number is30
PL/SQL procedure successfully completed.
